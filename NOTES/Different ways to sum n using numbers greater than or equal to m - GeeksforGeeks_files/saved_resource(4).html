          #include <bits stdc++.h="">
          #define MAX 100
          using namespace std;

          int dp[MAX][MAX];

          int numberofways(int n, int m)
          {
          if(dp[n][m] != -1)
          return dp[n][m];
          if(n == 0)
          return 1;
          if(m > n || n < 0)
          return 0;

          dp[n][m] = numberofways(n-m, m)+numberofways(n, m+1);

          return dp[n][m];
          }

                                              Kunal Khade Dipesh Agarwal • 2 years ago • edited

                                              import java.util.Arrays;

                                              public class SummationWays {

                                              public static int sumCount(int m, int n) {
                                              if(n < m) return 0;

                                              int table[]=new int[n+1];
                                              Arrays.fill(table, 0);
                                              table[0] =1;

                                              for(int i = m; i<= n; i++)
                                              for(int j=i; j<=n; j++)
                                              table[j] += table[j-i];

                                              return table[n];
                                              }

                                              public static void main(String args[]) {
                                              int m=2;
                                              int n=1;
                                              System.out.println( sumCount(m, n) );
                                              }
                                              }

                                              //it is important to return 0 when n < m
                                              //else output will be wrong
